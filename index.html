name: Refresh nursing job board # Runs the scraper on a schedule, regenerates jobs.json, and commits it back to # the repo. If this repo is connected to Netlify, the commit triggers a deploy # and the Slate page picks up the new feed automatically. on: schedule: - cron: "0 */6 * * *" # every 6 hours (UTC). Adjust as you like. workflow_dispatch: {} # lets you run it manually from the Actions tab permissions: contents: write # needed to commit the updated feed back jobs: refresh: runs-on: ubuntu-latest steps: - name: Check out repo uses: actions/checkout@v4 - name: Set up Python uses: actions/setup-python@v5 with: python-version: "3.11" - name: Install dependencies run: pip install requests beautifulsoup4 - name: Refresh listings and write feed env: USAJOBS_API_KEY: ${{ secrets.USAJOBS_API_KEY }} USAJOBS_EMAIL: ${{ secrets.USAJOBS_EMAIL }} run: python nurse_job_board.py --refresh --json --verbose - name: Commit updated feed and archive run: | git config user.name "job-board-bot" git config user.email "actions@users.noreply.github.com" git add jobs.json jobs.db if git diff --cached --quiet; then echo "No changes to commit." else git commit -m "Refresh job feed ($(date -u +'%Y-%m-%d %H:%M UTC'))" # Push, but if the remote moved while we were running (e.g. a web # edit or an overlapping run), pull --rebase and retry rather than # failing the whole run. for attempt in 1 2 3; do if git push; then echo "Pushed on attempt $attempt." break fi echo "Push rejected; syncing with remote and retrying ($attempt)..." git pull --rebase --autostash origin main || true sleep 3 done fi
